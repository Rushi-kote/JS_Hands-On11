Q7: Write a function that accepts parameter n and returns factorial of n"

function fact(n){
    let fact=1;
    if(n==1 || n==0){
        console.log(fact);
    }else{
        for(let i=2;i< =n;i++){
            fact*=i;
        }
        console.log(fact);
    }
}

fact(5);
